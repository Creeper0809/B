struct T {
  a: u8;
  b: u16;
  c: u32;
  d: u64;
};

func main() {
  var t: T;

  t.a = 1;
  t.b = 0x0203;
  t.c = 0x04050607;
  t.d = 9;

  // If sized loads/stores are correct:
  // - t.b & 255 == 3
  // - t.c & 255 == 7
  // Sum => 1 + 3 + 7 + 9 = 20
  return t.a + (t.b & 255) + (t.c & 255) + (t.d & 255);
}
